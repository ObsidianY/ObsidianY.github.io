2.6Test

Greedy Pie Eaters

这是一道我之前做过的 $dp$。这里直接

设 $dp[i][j]$ 表示区间 $[l,r]$ 间最大可能的重量；
$maxn[i][j][k]$ 表示吃的区间 $[l,r]$ 为 $[i, j]$ 的子区间且 $k$ 在 $[l,r]$ 中。

先处理出 $maxn$， 再递推处理 $dp$， $dp[i][j] = max(dp[i][k-1]+maxn[i][j][k]+dp[k+1][j])$

Bessie’s Snow Cow

这题可以利用树上只有 $n$ 个点用 set 直接做，但是我有一个分块做法。

用 $dfs$ 序把子树问题转成序列问题。

然后记 bool 维护状态。

具体很复杂，可以看代码：

// Author:ObsidianY

#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define ull unsigned long long
#define In inline
#define INF 0x3f3f3f3f
#define For(i, a, b) for (register int i = a; i <= b; ++i)
#define foR(i, a, b) for (register int i = a; i >= b; --i)
#define ms(a, b) memset(a, b, sizeof(a))
#define mc(a, b) memcpy(a, b, sizeof(b))
#define pb push_back
#define eb emplace_back
#define mp make_pair
#define sz size
#define fi first
#define se second
#define pii pair<int, int>
#define debug(x) cerr << #x << " = " << x << endl
#define err cerr << "Error\n"
#define GT cerr << 1.000 * clock() / CLOCKS_PER_SEC << endl;
#define file(FILE_NAME) freopen(FILE_NAME ".in", "r", stdin), freopen(FILE_NAME ".out", "w", stdout)
#define filein(FILE_NAME) freopen(FILE_NAME ".in", "r", stdin)
#define fileout(FILE_NAME) freopen(FILE_NAME ".out", "w", stdout)

const int N = 1e5 + 10, sqN = 500;
int n, m;

int cnt, head[N];

struct Edge {
    int to, nxt;
} edge[N << 1];

In int read() {
    int x = 0, f = 1, ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') x = x * 10 + ch - 48, ch = getchar();
    return x * f;
}

In void AddEdge(int u, int v) {
    edge[++cnt] = (Edge){ v, head[u] };
    head[u] = cnt;
    return;
}

int tim, dfn[N], siz[N];

void dfs(int u, int fa) {
    dfn[u] = ++tim, siz[u] = 1;
    for (int i = head[u], v; i; i = edge[i].nxt) {
        v = edge[i].to;
        if (v == fa)
            continue;
        dfs(v, u);
        siz[u] += siz[v];
    }
    return;
}

int bnum, st[sqN], ed[sqN], bl[N], sta[sqN][N];  // 0: simple; -1: all; ortherwise: the number is the new id
bool used[sqN][N << 1];
LL allnum[sqN], pre[N];

void build() {
    bnum = sqrt(n);
    For(i, 1, bnum) st[i] = ed[i - 1] + 1, ed[i] = n / bnum * i;
    ed[bnum] = n;
    For(i, 1, bnum) For(j, st[i], ed[i]) bl[j] = i;
    return;
}

In void mdls(int id, int cur, int l, int r) {
    int delta = 0;
    For(i, l - st[id] + 1, r - st[id] + 1) {
        if (!used[i][cur])
            used[i][cur] = 1, ++delta;
        pre[i + st[id] - 1] += delta;
    }
    For(i, r + 1, ed[id]) pre[i] += delta;
    return;
}

In void dealls(int l, int r, int c) {
    int id = bl[l];
    if (sta[id][c] == -1)
        return;
    if (sta[id][c] != 0)
        mdls(id, sta[id][c], l, r);
    else {
        sta[id][c] = ++cnt;
        mdls(id, cnt, l, r);
    }
    return;
}

void modify(int l, int r, int c) {
    if (bl[l] == bl[r])
        return dealls(l, r, c);
    dealls(l, ed[bl[l]], c), dealls(st[bl[r]], r, c);
    For(i, bl[l] + 1, bl[r] - 1) {
        if (sta[i][c] > 0) {
            int delta = 0, cur = sta[i][c];
            For(j, 1, ed[i] - st[i] + 1) {
                if (used[j][cur])
                    ++delta;
                pre[j + st[i] - 1] -= delta;
            }
        }
        if (sta[i][c] != -1)
            ++allnum[i], sta[i][c] = -1;
    }
    return;
}

LL calls(int l, int r) {
    return 1ll * (r - l + 1) * allnum[bl[l]] + pre[r] - (l == st[bl[l]] ? 0 : pre[l - 1]);
}

LL query(int l, int r) {
    if (bl[l] == bl[r]) {
        return calls(l, r);
    }
    LL res = 0;
    res += calls(l, ed[bl[l]]) + calls(st[bl[r]], r);
    For(i, bl[l] + 1, bl[r] - 1) res += pre[ed[i]] + 1ll * allnum[i] * (ed[i] - st[i] + 1);
    return res;
}

signed main() {
    // file("snowcow");
    n = read(), m = read();
    For(i, 1, n - 1) {
        int a = read(), b = read();
        AddEdge(a, b), AddEdge(b, a);
    }
    dfs(1, 0);
    cnt = 0;
    build();
    // For(i, 1, n) cerr<<dfn[i]<<" "<<siz[i]<<" "<<bl[i]<<endl;
    int opt, x, c, l, r;
    For(i, 1, m) {
        opt = read(), x = read();
        l = dfn[x], r = dfn[x] + siz[x] - 1;
        if (opt == 1) {
            c = read();
            modify(l, r, c);
        } else
            printf("%lld\n", query(l, r));
    }
#define _ 0
    return 0 ^ _ ^ 0;
#undef _
}

Tree Depth

这是一棵笛卡尔树。

把区间拆开，然后对增加的逆序对个数 $dp$。

发现 $dp$ 转移就是乘以一个 $x^0+x^1+\ldots+x^d$ 的生成函数。乘法 $O(k)$ 实现。

预处理所有生成函数的值，枚举 $j-i$ 就行了。
